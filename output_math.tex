\textbf{TRIGONOMETRY (10th Class)}

\textbf{Exercise 8.1}

\textbf{Quest1}: In $\triangle ABC$, right angle at $B$, $AB = 24$ cm, $BC = 7$ cm. Determine:
(i) $\sin A$, $\cos A$
(ii) $\sin C$, $\cos C$

\textbf{Sol}: Let us draw a triangle $\triangle ABC$.

By using the Pythagoras theorem, we have

\begin{equation}
AC^2 = AB^2 + BC^2
\end{equation}

\begin{equation}
AC^2 = (24)^2 + (7)^2
\end{equation}

\begin{equation}
AC^2 = 576 + 49
\end{equation}

\begin{equation}
AC^2 = 625
\end{equation}

\begin{equation}
AC = \sqrt{625} = 25 \text{ cm}.
\end{equation}

(i) $\sin A = \frac{BC}{AC} = \frac{7}{25}$ [since $\sin \theta = \frac{P}{H}$]

\begin{equation}
\cos A = \frac{AB}{AC} = \frac{24}{25} \quad [\text{since} \; \cos \theta = \frac{B}{H}]
\end{equation}

(ii) $\sin C = \frac{AB}{AC} = \frac{24}{25}$ [since $\sin \theta = \frac{P}{H}$]

\begin{equation}
\cos C = \frac{BC}{AC} = \frac{7}{25} \quad [\text{since} \; \cos \theta = \frac{B}{H}]
\end{equation}

\textbf{Quest2}: In adjoining figure, find $\tan P - \cot R$.

\textbf{Sol}: By using the Pythagoras theorem, we have

\begin{equation}
PR^2 = PQ^2 + QR^2
\end{equation}

\begin{equation}
(13)^2 = (12)^2 + QR^2
\end{equation}

\begin{equation}
QR^2 = (13)^2 - (12)^2
\end{equation}
